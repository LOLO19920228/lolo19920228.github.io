require、module exports 模組設計

// data.js
let data = 2;
module.exports = {
  content: data,
  title: 'title'
};
exports.ddd = 2;

// app.js
let content = require('./data');
console.log(content);
console.log(content.ddd); // 會被module.exports覆蓋掉
